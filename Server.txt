import java.io.*;
import java.net.*;
import javax.swing.*;
import java.util.*;
import java.awt.*;

/**
 * Main (Server) - Java Messenger App
 * Accepts multiple client connections and handles broadcast + private messaging.
 * Includes a server-side monitor GUI to track all activity.
 */
public class Main {

    private static final int PORT = 5555;

    private static Set<PrintWriter> clientWriters = Collections.synchronizedSet(new HashSet<>());
    private static Set<String> clientNames = Collections.synchronizedSet(new HashSet<>());

    static JTextArea monitorArea;
    static JTextArea onlineUsersArea;

    public static void main(String[] args) {

        // ─── Server Monitor Window ───────────────────────────────────────────
        SwingUtilities.invokeLater(() -> {
            JFrame monitorFrame = new JFrame("🖥️ Server Monitor");
            monitorFrame.setLayout(null);
            monitorFrame.setBounds(150, 150, 540, 540);
            monitorFrame.getContentPane().setBackground(new Color(20, 20, 20));

            JLabel heading = new JLabel("MONITOR MODE");
            heading.setBounds(20, 8, 300, 40);
            heading.setFont(new Font("Monospaced", Font.BOLD, 22));
            heading.setForeground(new Color(0, 200, 100));
            monitorFrame.add(heading);

            monitorArea = new JTextArea();
            monitorArea.setFont(new Font("Monospaced", Font.PLAIN, 13));
            monitorArea.setEnabled(false);
            monitorArea.setBackground(new Color(15, 15, 15));
            monitorArea.setForeground(new Color(200, 200, 200));
            monitorArea.setDisabledTextColor(new Color(200, 200, 200));
            monitorArea.setBorder(BorderFactory.createEmptyBorder(8, 10, 8, 10));
            monitorArea.setLineWrap(true);

            JScrollPane scrollV = new JScrollPane(monitorArea);
            scrollV.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
            scrollV.setBounds(10, 55, 510, 460);
            scrollV.setBorder(BorderFactory.createLineBorder(new Color(60, 60, 60)));
            monitorFrame.add(scrollV);

            monitorFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            monitorFrame.setVisible(true);
        });

        // ─── Online Users Window ─────────────────────────────────────────────
        SwingUtilities.invokeLater(() -> {
            JFrame onlineFrame = new JFrame("🟢 Connected Clients");
            onlineFrame.setLayout(null);
            onlineFrame.setBounds(720, 150, 260, 400);
            onlineFrame.getContentPane().setBackground(new Color(25, 25, 25));

            JLabel label = new JLabel("Connected Clients");
            label.setBounds(15, 8, 220, 35);
            label.setForeground(new Color(0, 200, 100));
            label.setFont(new Font("Monospaced", Font.BOLD, 14));
            onlineFrame.add(label);

            onlineUsersArea = new JTextArea();
            onlineUsersArea.setBounds(0, 50, 260, 330);
            onlineUsersArea.setBackground(new Color(25, 25, 25));
            onlineUsersArea.setForeground(new Color(200, 200, 200));
            onlineUsersArea.setFont(new Font("Monospaced", Font.PLAIN, 13));
            onlineUsersArea.setEnabled(false);
            onlineUsersArea.setDisabledTextColor(new Color(200, 200, 200));
            onlineUsersArea.setBorder(BorderFactory.createEmptyBorder(8, 12, 8, 8));
            onlineFrame.add(onlineUsersArea);

            onlineFrame.setVisible(true);
            onlineFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        });

        // ─── Start Server ────────────────────────────────────────────────────
        System.out.println("[Server] Started on port " + PORT);

        try (ServerSocket serverSocket = new ServerSocket(PORT)) {
            while (true) {
                Socket clientSocket = serverSocket.accept();
                String clientName = clientSocket.getInetAddress().getHostName();
                log("[+] Connected: " + clientSocket.getRemoteSocketAddress());
                new ClientHandler(clientSocket, clientName).start();
            }
        } catch (Exception e) {
            log("[ERROR] Server crashed: " + e.getMessage());
        }
    }

    /** Append a log message to the monitor GUI */
    static void log(String message) {
        System.out.println(message);
        SwingUtilities.invokeLater(() -> {
            if (monitorArea != null) {
                monitorArea.append(message + "\n");
                monitorArea.setCaretPosition(monitorArea.getDocument().getLength());
            }
        });
    }

    /** Broadcast a message to all connected clients */
    static void broadcast(String message) {
        synchronized (clientWriters) {
            for (PrintWriter writer : clientWriters) {
                writer.println(message);
            }
        }
    }

    /** Update the online users list on server GUI and notify all clients */
    static void updateOnlineClients() {
        StringBuilder onlineUsers = new StringBuilder("ONLINE_USERS:");
        synchronized (clientNames) {
            for (String name : clientNames) {
                onlineUsers.append(name).append(",");
            }
        }

        String userList = onlineUsers.toString();
        synchronized (clientWriters) {
            for (PrintWriter writer : clientWriters) {
                writer.println(userList);
            }
        }

        SwingUtilities.invokeLater(() -> {
            if (onlineUsersArea != null) {
                onlineUsersArea.setText("");
                synchronized (clientNames) {
                    for (String name : clientNames) {
                        onlineUsersArea.append("🟢 " + name + "\n");
                    }
                }
            }
        });
    }

    // ─── Client Handler Thread ────────────────────────────────────────────────
    static class ClientHandler extends Thread {

        private final Socket socket;
        private final String clientName;
        private PrintWriter out;
        private BufferedReader in;

        public ClientHandler(Socket socket, String clientName) {
            this.socket = socket;
            this.clientName = clientName;
        }

        @Override
        public void run() {
            try {
                in  = new BufferedReader(new InputStreamReader(socket.getInputStream()));
                out = new PrintWriter(socket.getOutputStream(), true);

                synchronized (clientWriters) { clientWriters.add(out); }
                synchronized (clientNames)   { clientNames.add(clientName); }
                updateOnlineClients();

                String message;
                while ((message = in.readLine()) != null) {
                    log("[" + clientName + "]: " + message);

                    if (message.startsWith("@")) {
                        // Private message: @username: message
                        int colon = message.indexOf(":");
                        if (colon != -1) {
                            String targetUser  = message.substring(1, colon).trim();
                            String privateMsg  = message.substring(colon + 1).trim();
                            sendPrivateMessage(targetUser, privateMsg);
                        }
                    } else {
                        broadcast(clientName + ": " + message);
                    }
                }

            } catch (Exception e) {
                log("[!] Client error (" + clientName + "): " + e.getMessage());
            } finally {
                disconnect();
            }
        }

        /** Send a private message to a specific user */
        private void sendPrivateMessage(String targetUser, String message) {
            synchronized (clientNames) {
                boolean found = false;
                for (String name : clientNames) {
                    if (name.equalsIgnoreCase(targetUser)) {
                        found = true;
                        // Note: In a production app, maintain a name→writer map for O(1) lookup
                        synchronized (clientWriters) {
                            for (PrintWriter writer : clientWriters) {
                                writer.println("[Private from " + clientName + "]: " + message);
                            }
                        }
                        break;
                    }
                }
                if (!found) {
                    out.println("[System]: User '" + targetUser + "' is not online.");
                }
            }
        }

        /** Clean up on disconnect */
        private void disconnect() {
            try {
                socket.close();
            } catch (IOException ignored) {}

            synchronized (clientWriters) { clientWriters.remove(out); }
            synchronized (clientNames)   { clientNames.remove(clientName); }

            log("[-] Disconnected: " + clientName);
            updateOnlineClients();
        }
    }
}
