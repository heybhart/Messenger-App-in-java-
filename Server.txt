    private static class ClientHandler extends Thread
    {
        private Socket socket;
        private PrintWriter out;
        private BufferedReader in;
        private String clientName;
        private JTextArea textarea2;

        public ClientHandler(Socket socket, JTextArea area , String clientname , JTextArea textarea2)
        {
            this.socket = socket;
            this.clientName = clientname;
            this.textarea2 = textarea2;
        }

        public void run()
        {
            try
            {
                in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
                out = new PrintWriter(socket.getOutputStream(), true);

                synchronized(clientWriters)
                {
                    clientWriters.add(out); // Add Output Stream of every client in ClientWriter , for send the message to everyone whoever is connected
                }

                synchronized(clientNames)
                {
                    clientNames.add(clientName); // Add client name
                    updateOnlineClients(); // Update the list of clients
                }
                String message;
                while ((message = in.readLine()) != null) {
                    System.out.println("Received: " + message);
                    area.append(this.clientName + ": " + message + "\n");

                    if (message.startsWith("@")) {
                        // Extract the target username and the actual message
                        int colonIndex = message.indexOf(":");
                        if (colonIndex != -1) {
                            String targetUser = message.substring(1, colonIndex).trim(); // Username between '@' and ':'
                            String privateMessage = message.substring(colonIndex + 1).trim(); // Message after ':'

                            // Send message only to the specified user
                            sendPrivateMessage(targetUser, privateMessage);
                        }
                    } else {
                        broadcast(message); // Regular message broadcast
                    }
                }
//                String message;
//                while ((message = in.readLine()) != null)
//                {
//                    System.out.println("Received: " + message);
//                    area.append(this.clientName + ": " + message + "\n");
//                    broadcast(message);
//                }
            }
            catch (Exception e)
            {
                System.out.println(e);
            }
            finally
            {
                try
                {
                    socket.close();
                }
                catch (Exception e)
                {
                    System.out.println(e);
                }
//                synchronized(clientWriters)
//                {
//                    clientWriters.remove(out);
//                }

                synchronized(clientNames)
                {
                    clientNames.remove(clientName); // Remove the client on disconnect
                    updateOnlineClients(); // Update the list of clients
                }
            }
        }
        private PrintWriter findWriterForUser(String targetUser) {
            synchronized(clientWriters) {
                for (PrintWriter writer : clientWriters) {
                    if (clientNames.contains(targetUser)) {
                        return writer;
                    }
                }
            }
            return null;
        }
        private void sendPrivateMessage(String targetUser, String message) {
            // Check if the target user is connected
            synchronized (clientNames) {
                boolean userFound = false;                  // To check if user exists
                for (String name : clientNames) {         // Check if the target username matches the connected client name
                    if (name.contains(targetUser)) {
                        userFound = true;
                        PrintWriter writer = findWriterForUser(name);
                        if (writer != null) {
                            writer.println("Private from " + this.clientName + ": " + message); // Send private message
                        }
                        break; // Exit loop after sending the message
                    }
                }
                if (!userFound) {
                    out.println("User " + targetUser + " is not online."); // Notify sender if user not found
                }
            }
        }
        private void broadcast(String message)
        {
            synchronized (clientWriters)
            {
                for (PrintWriter writer : clientWriters)
                {
                    writer.println(message);
                }
            }
        }
        private void updateOnlineClients() {
            synchronized(clientWriters) {
                StringBuilder onlineUsers = new StringBuilder("ONLINE_USERS:");
                for (String name : clientNames) {
                    onlineUsers.append(name).append(",");
                }
                String userList = onlineUsers.toString();

                for (PrintWriter writer : clientWriters) {
                    writer.println(userList); // Send the online users list to each client
                }

                // Server-side update of online users textarea
                SwingUtilities.invokeLater(() -> {
                    textarea2.setText("");
                    for (String name : clientNames) {
                        textarea2.append(name + "  "+"🟢\n"); // Add each client name with green dot
                    }
                });
            }
        }
