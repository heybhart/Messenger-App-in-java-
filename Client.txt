import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.*;
import java.net.*;

/**
 * Client - Java Messenger App
 * Connects to the server and provides a GUI for sending/receiving messages.
 * Supports broadcast and private messaging (@username: message).
 */
public class Client {

    private static final String SERVER_ADDRESS = "127.0.0.1";
    private static final int PORT = 5555;

    public static void main(String[] args) {

        // ─── Main Chat Window ───────────────────────────────────────────────
        JFrame chatFrame = new JFrame("Messenger — Client");
        chatFrame.setBounds(300, 150, 420, 620);
        chatFrame.setLayout(null);
        chatFrame.getContentPane().setBackground(new Color(30, 30, 30));

        JLabel headerLabel = new JLabel("💬 Messenger");
        headerLabel.setBounds(20, 5, 300, 45);
        headerLabel.setFont(new Font("Monospaced", Font.BOLD, 20));
        headerLabel.setForeground(new Color(0, 200, 100));
        chatFrame.add(headerLabel);

        JTextArea chatArea = new JTextArea();
        chatArea.setFont(new Font("Monospaced", Font.PLAIN, 14));
        chatArea.setEditable(false);
        chatArea.setBackground(new Color(20, 20, 20));
        chatArea.setForeground(new Color(220, 220, 220));
        chatArea.setLineWrap(true);
        chatArea.setWrapStyleWord(true);
        chatArea.setBorder(BorderFactory.createEmptyBorder(8, 8, 8, 8));

        JScrollPane scrollPane = new JScrollPane(chatArea);
        scrollPane.setBounds(10, 55, 390, 460);
        scrollPane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
        scrollPane.setBorder(BorderFactory.createLineBorder(new Color(60, 60, 60)));
        chatFrame.add(scrollPane);

        JTextField msgInput = new JTextField();
        msgInput.setBounds(10, 530, 290, 38);
        msgInput.setBackground(new Color(45, 45, 45));
        msgInput.setForeground(Color.WHITE);
        msgInput.setCaretColor(Color.WHITE);
        msgInput.setFont(new Font("Monospaced", Font.PLAIN, 13));
        msgInput.setBorder(BorderFactory.createCompoundBorder(
                BorderFactory.createLineBorder(new Color(0, 200, 100)),
                BorderFactory.createEmptyBorder(4, 8, 4, 8)
        ));
        chatFrame.add(msgInput);

        JButton sendBtn = new JButton("Send ➤");
        sendBtn.setBounds(310, 530, 100, 38);
        sendBtn.setBackground(new Color(0, 150, 80));
        sendBtn.setForeground(Color.WHITE);
        sendBtn.setFocusPainted(false);
        sendBtn.setFont(new Font("Monospaced", Font.BOLD, 13));
        sendBtn.setBorderPainted(false);
        sendBtn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
        chatFrame.add(sendBtn);

        JLabel hintLabel = new JLabel("Tip: Use @username: message for private chat");
        hintLabel.setBounds(10, 575, 390, 20);
        hintLabel.setFont(new Font("Monospaced", Font.PLAIN, 11));
        hintLabel.setForeground(new Color(120, 120, 120));
        chatFrame.add(hintLabel);

        // ─── Online Users Window ─────────────────────────────────────────────
        JFrame onlineFrame = new JFrame("🟢 Online Users");
        onlineFrame.setLayout(null);
        onlineFrame.setBounds(750, 150, 260, 400);
        onlineFrame.getContentPane().setBackground(new Color(25, 25, 25));

        JLabel onlineHeader = new JLabel("Online Now");
        onlineHeader.setBounds(15, 8, 200, 35);
        onlineHeader.setForeground(new Color(0, 200, 100));
        onlineHeader.setFont(new Font("Monospaced", Font.BOLD, 15));
        onlineFrame.add(onlineHeader);

        JTextArea onlineUsersArea = new JTextArea();
        onlineUsersArea.setBounds(0, 50, 260, 330);
        onlineUsersArea.setBackground(new Color(25, 25, 25));
        onlineUsersArea.setForeground(new Color(200, 200, 200));
        onlineUsersArea.setFont(new Font("Monospaced", Font.PLAIN, 13));
        onlineUsersArea.setEnabled(false);
        onlineUsersArea.setBorder(BorderFactory.createEmptyBorder(8, 12, 8, 8));
        onlineFrame.add(onlineUsersArea);

        onlineFrame.setVisible(true);
        onlineFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // ─── Socket Connection ───────────────────────────────────────────────
        try {
            Socket socket = new Socket(SERVER_ADDRESS, PORT);
            BufferedReader in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            PrintWriter out = new PrintWriter(socket.getOutputStream(), true);

            // Send button action
            ActionListener sendAction = e -> {
                String msgToSend = msgInput.getText().trim();
                if (!msgToSend.isEmpty()) {
                    out.println(msgToSend);
                    chatArea.append("  Me: " + msgToSend + "\n");
                    msgInput.setText("");
                    chatArea.setCaretPosition(chatArea.getDocument().getLength());
                }
            };

            sendBtn.addActionListener(sendAction);
            msgInput.addActionListener(sendAction); // Press Enter to send

            // Incoming message listener thread
            new Thread(() -> {
                String receivedMsg;
                try {
                    while ((receivedMsg = in.readLine()) != null) {
                        if (receivedMsg.startsWith("ONLINE_USERS:")) {
                            String[] users = receivedMsg.substring("ONLINE_USERS:".length()).split(",");
                            SwingUtilities.invokeLater(() -> {
                                onlineUsersArea.setText("");
                                for (String user : users) {
                                    if (!user.trim().isEmpty()) {
                                        onlineUsersArea.append("🟢 " + user.trim() + "\n");
                                    }
                                }
                            });
                        } else {
                            final String msg = receivedMsg;
                            SwingUtilities.invokeLater(() -> {
                                chatArea.append("  " + msg + "\n");
                                chatArea.setCaretPosition(chatArea.getDocument().getLength());
                            });
                        }
                    }
                } catch (IOException ex) {
                    SwingUtilities.invokeLater(() ->
                            chatArea.append("\n[Disconnected from server]\n"));
                }
            }).start();

        } catch (IOException ex) {
            JOptionPane.showMessageDialog(chatFrame,
                    "Could not connect to server at " + SERVER_ADDRESS + ":" + PORT,
                    "Connection Error", JOptionPane.ERROR_MESSAGE);
        }

        chatFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        chatFrame.setVisible(true);
    }
}
